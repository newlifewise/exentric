Sample: dhtmlxMenu Add/delete menu items dhtmlxMenu main page
 
  • Add new image button  
  • click on any existing button for deleting it
    		•  

     
<div id="menu_zone" style="width:600; height:50;"/>
<script>
     
        //init menu
        aMenuBar=new dhtmlXMenuBarObject('menu_zone','100%',30,"Demo menu");
        // set processing function
        aMenuBar.setOnClickHandler(onButtonClick);
        //load menu from xml
        aMenuBar.loadXML("_menu.xml")
        //show menu
        aMenuBar.showBar();
        
        //add new button
        aMenuBar.addItem(parentPanel,
            new dhtmlXMenuItemObject(newId,newName, width, imgsrc))
        //delete button 
        aMenuBar.removeItem(itemId);
    </script>

<div id="menu_zone" style="width:600;"/>
<script>
 
    //init menu
    aMenuBar=new dhtmlXMenuBarObject('menu_zone','100%',30,"Demo menu");
    // set processing function
    aMenuBar.setOnClickHandler(onButtonClick);
    //load menu from xml
    aMenuBar.loadXML("_menu.xml")
    //show menu
    aMenuBar.showBar();
    //show item
    aMenuBar.showItem(itemId);
    //hide item
    aMenuBar.hideItem(itemId);
    //disable item
    aMenuBar.disableItem(itemId);    
    //enable item
    aMenuBar.enableItem(itemId);
</script>
